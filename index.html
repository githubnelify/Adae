void selectionSort(int n) {
int i, j, min_idx, t;
for(i = 0; i < n; i++) {
min_idx = i;
for(j = i + 1; j < n; j++) {
if (numList[j] < numList[min_idx])
min_idx = j;
}
if (min_idx != i) {
t = numList[i];
numList[i] = numList[min_idx];
numList[min_idx] = t;
}
}
}

void merge(int low, int mid, int high) {
vector<int> b;
int i = low;
int j = mid + 1;
int k = 0;
while (i <= mid && j <= high) {
if (numList[i] < numList[j])
b.push_back(numList[i++]);
else
b.push_back(numList[j++]);
}
while (i <= mid)
b.push_back(numList[i++]);
while (j <= high)
b.push_back(numList[j++]);
for (i = low; i <= high; i++)
numList[i] = b[k++];
}
void SortArray(int low, int high) {
if (low < high) {
int mid = (low + high) / 2;
SortArray(low, mid);
SortArray(mid + 1, high);
merge(low, mid, high);
}
}
int main() {
int n;

#include <iostream>
#include <vector>

using namespace std;
vector<int> numList;
void fnDispArray(int n) {

cout << numList[i] << endl;
}
void fnSwap(int &a, int &b) {
int t = a;
a = b;
b = t;
}
void fnGenRandArray(int n) {
srand(time(NULL));
for (int i = 0; i < n; i++) {
int val = rand() % 10000 + 5000;
numList.push_back(val);
}
}
int fnPartition(int l, int r) {
int i = l;
int j = r + 1;
int p = numList[l];
do {
do { i++; } while (i <= r && numList[i] < p);
do { j--; } while (numList[j] > p);
if (i < j) fnSwap(numList[i], numList[j]);
} while (i < j);
fnSwap(numList[l], numList[j]);
return j;
}
void fnQuickSort(int l, int r) {
if (l < r) {
int s = fnPartition(l, r);
fnQuickSort(l, s - 1);
fnQuickSort(s + 1, r);
}
}
int main() {
int n;
cout << "\nEnter number of elements to sort: ";

fnGenRandArray(n);
cout << "Unsorted array: \n";

9b.warshall
#include <iostream>
using namespace std;
const int MAX = 100;
void Warshall(int graph[MAX][MAX], int n) {
int i, j, k;
for (k = 0; k < n; k++) {
for (i = 0; i < n; i++) {
for (j = 0; j < n; j++) {
if (graph[i][j] || (graph[i][k] && graph[k][j])) {
graph[i][j] = 1;
}
}
}
}
}
int main() {
int i, j, n;
int graph[MAX][MAX];
cout << "Enter the number of vertices: ";
cin >> n;
cout << "Enter the adjacency matrix:\n";
for (i = 0; i < n; i++) {
for (j = 0; j < n; j++) {
cin >> graph[i][j];
}
}
Warshall(graph, n);
cout << "\nThe transitive closure for the given graph is:\n";
for (i = 0; i < n; i++) {
for (j = 0; j < n; j++) {
cout << graph[i][j] << "\t";
}
cout << endl;
}
return 0;
}

#include <iostream>
#include <vector>
using namespace std;
bool flag = false;
vector<int> subset;
void sumofsubset(int i, int n, int set[], int Sum, vector<int> subset) {
if (Sum == 0) {
flag = true;
cout << "[";
for (int i = 0; i < subset.size(); i++) {
cout << subset[i] << " ";
}
cout << "]" << endl;
return;
}
if (i >= n) {
return;
}
sumofsubset(i + 1, n, set, Sum, subset);
if (set[i] <= Sum) {
subset.push_back(set[i]);
sumofsubset(i + 1, n, set, Sum - set[i], subset);
subset.pop_back();
}
}
int main() {
int set[] = {5, 10, 12, 13, 15, 18};
int sum = 30;
int n = sizeof(set) / sizeof(set[0]);
sumofsubset(0, n, set, sum, subset);
if (!flag) {

}
return 0;
}
8.Dijkstra's algorithm.
#include <iostream>
#include <limits.h>
using namespace std;
const int MAXNODES = 10;
const int INF = INT_MAX;
void fnDijkstra(int c[MAXNODES][MAXNODES], int d[MAXNODES], int P[MAXNODES], int
s[MAXNODES], int so, int n) {
int i, j, a, b, min;
for (i = 0; i < n; i++) {
s[i] = 0;
d[i] = c[so][i];
P[i] = so;
}
s[so] = 1;
for (i = 1; i < n; i++) {
min = INF;
a = -1;
for (j = 0; j < n; j++) {
if (s[j] == 0 && d[j] < min) {
min = d[j];
a = j;
}
}
if (a == -1) return;
s[a] = 1;
for (b = 0; b < n; b++) {
if (s[b] == 0 && d[a] + c[a][b] < d[b]) {
d[b] = d[a] + c[a][b];
P[b] = a;
}
}
}
}
int main() {
int n, cost[MAXNODES][MAXNODES], dist[MAXNODES], visited[MAXNODES],
path[MAXNODES], i, j, source, dest;
cout << "\nEnter the number of nodes: ";
cin >> n;
cout << "Enter the Cost Matrix:" << endl;
for (i = 0; i < n; i++) {
for (j = 0; j < n; j++) {
cin >> cost[i][j];
}
}
cout << "Enter the Source vertex: ";
cin >> source;
for (i = 0; i < n; i++) {
dist[i] = INF;
path[i] = -1;
visited[i] = 0;
}
dist[source] = 0;
cout << "\nFor Source Vertex: " << source << " shortest path to other
vertices:" << endl;
for (dest = 0; dest < n; dest++) {
if (dest != source) {
fnDijkstra(cost, dist, path, visited, source, n);
}
if (dist[dest] == INF) {
cout << dest << " not reachable" << endl;
} else {
cout << dest << " = " << dist[dest] << endl;
}
}
return 0;
}
